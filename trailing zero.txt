void main() {
  int number = 30
    ; 
  int factorial = 1;
    for (int i = 1; i <= number; i++) {
      factorial *= i;
    }
  int count = 0;
  for (int i = 5; i <= number; i *= 5) {
    count += number ~/ i;
  }
  print("Factorial of $number is $factorial");
  print("Trailing zeros in $number! = $count");

}
