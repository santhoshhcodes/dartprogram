void main(){
  int num = 145;
  int temp = num; 
  int sum = 0;
  
  for(int i =num; i>0 ; i~/=10){
    int digits = i%10;
    int fact = 1;
    for(int j = 1 ; j <= digits ; j++){
      fact *= j;
      
    }sum += fact;
   
  } if(sum == temp){
      print("$temp is a strong number");
    }else{
      print("$temp is not strong number ");
    }
}
